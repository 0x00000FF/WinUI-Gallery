private async void ShowDialog_Click(object sender, RoutedEventArgs e)
{
    ContentDialog dialog = new()
    {
        Title = "Save your work?",
        PrimaryButtonText = "Save",
        SecondaryButtonText = "Don't Save",
        CloseButtonText = "Cancel",
        DefaultButton = ContentDialogButton.Primary,
        // XamlRoot must be set in WinUI 3.
        // Any UI element belongs to a visual tree whose root is XamlRoot.
        XamlRoot = (sender as Button)?.XamlRoot,
        Content = new ContentDialogContent()
    };
    var result = await dialog.ShowAsync();
    // result contains whether the user clicked the primary button,
    // the secondary button, or the close button
}
